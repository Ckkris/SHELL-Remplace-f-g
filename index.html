Shell-remplace-f-g

Remplacer les lettres f par g dans les noms de fichiers textes ayant plus de 10 lignes

Download this project as a .zip file Download this project as a tar.gz file

!/bin/sh

for i in $( find . -name 'f' ) do #echo $i count=$( cat $i | wc -l) #echo $count if [ $count -gt 10 ] then echo "$i;$count" r=$( echo $i | sed 's/f/g/g') mv $i $r fi done
